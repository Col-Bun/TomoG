// ===== NUMOGRAM LOGIC & TIME SORCERY ===== function selectZone(zoneNum) { document.querySelectorAll('.numo-node').forEach(n => n.classList.remove('active')); const btn = document.getElementById('btn-zone-' + zoneNum); if(btn) btn.classList.add('active'); const dataHtml = document.getElementById('data-zone-' + zoneNum).innerHTML; document.getElementById('numo-display').innerHTML = dataHtml; } // Connections based on CCRU Lore // Syzygies (sum to 9): 9-0, 8-1, 7-2, 6-3, 5-4 // Tracts/Gates (descending currents): 2-3, 4-5, 7-8, 1-9 const numogramPaths = [ { from: 9, to: 0, type: 'syzygy' }, { from: 8, to: 1, type: 'syzygy' }, { from: 7, to: 2, type: 'syzygy' }, { from: 6, to: 3, type: 'syzygy' }, { from: 5, to: 4, type: 'syzygy' }, { from: 2, to: 3, type: 'tract' }, { from: 4, to: 5, type: 'tract' }, { from: 7, to: 8, type: 'tract' }, { from: 1, to: 9, type: 'tract' } ]; function drawNumogramLines() { const svg = document.getElementById('numo-svg'); if(!svg) return; svg.innerHTML = ''; // clear existing const container = document.getElementById('numo-map-container'); const rect = container.getBoundingClientRect(); numogramPaths.forEach(path => { const node1 = document.getElementById('btn-zone-' + path.from); const node2 = document.getElementById('btn-zone-' + path.to); if(node1 && node2) { // Get positions relative to the container const x1 = node1.offsetLeft + (node1.offsetWidth / 2); const y1 = node1.offsetTop + (node1.offsetHeight / 2); const x2 = node2.offsetLeft + (node2.offsetWidth / 2); const y2 = node2.offsetTop + (node2.offsetHeight / 2); const line = document.createElementNS('http://www.w3.org/2000/svg', 'line'); line.setAttribute('x1', x1); line.setAttribute('y1', y1); line.setAttribute('x2', x2); line.setAttribute('y2', y2); line.setAttribute('class', `numo-line ${path.type}`); svg.appendChild(line); } }); } // Draw lines on load, resize, and when the Numogram tab is clicked window.addEventListener('resize', drawNumogramLines); document.querySelector('[data-tab="numogram"]').addEventListener('click', () => { setTimeout(drawNumogramLines, 50); // slight delay to allow display:block to render }); // Gematria & Digital Root Calculator function calculateNumogram() { const input = document.getElementById('numo-gematria-input').value.trim().toUpperCase(); const display = document.getElementById('numo-result-display'); if (!input) { display.textContent = "Please enter data."; return; } let totalSum = 0; let calculationSteps = []; // Alphanumeric evaluation (A=1...Z=26) + numbers for (let i = 0; i < input.length; i++) { const char = input[i]; if (/[A-Z]/.test(char)) { const val = char.charCodeAt(0) - 64; // A is 65 totalSum += val; calculationSteps.push(`${char}(${val})`); } else if (/[0-9]/.test(char)) { const val = parseInt(char); totalSum += val; calculationSteps.push(val); } } if (totalSum === 0) { display.textContent = "No valid numeric or alphanumeric characters found."; return; } // Digital Root calculation (modulo 9 logic, but 9 remains 9 instead of 0) let digitalRoot = totalSum % 9; if (digitalRoot === 0 && totalSum > 0) digitalRoot = 9; // CCRU specific: sometimes 9 collapses to 0. We'll leave it as mathematical digital root for now. display.innerHTML = `Sum: ${totalSum} ➔ Digital Root: ${digitalRoot}`; // Trigger the map node selectZone(digitalRoot); } // Allow pressing enter in the input box document.getElementById('numo-gematria-input').addEventListener('keydown', function(e) { if (e.key === 'Enter') calculateNumogram(); }); // Initialize lines on first load just in case setTimeout(drawNumogramLines, 500);
